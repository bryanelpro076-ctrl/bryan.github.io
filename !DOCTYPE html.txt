    <script>
        const pantallaSobre = document.getElementById('pantalla-sobre');
        const contenedor = document.getElementById('contenedor');
        const celebracion = document.getElementById('celebracion');
        const btnSi = document.getElementById('btn-si');
        const btnNo = document.getElementById('btn-no');
        const heartRainContainer = document.getElementById('heart-rain-container');

        // Abrir sobre
        pantallaSobre.addEventListener('click', () => {
            pantallaSobre.classList.add('fade-out');
            setTimeout(() => {
                pantallaSobre.classList.add('hidden');
                contenedor.classList.add('visible');
            }, 500);
        });

        // Boton No que escapa
        let sizeInc = 0;
        btnNo.addEventListener('mouseover', () => {
            sizeInc++;
            const x = Math.random() * (window.innerWidth - btnNo.offsetWidth);
            const y = Math.random() * (window.innerHeight - btnNo.offsetHeight);
            btnNo.style.position = "fixed";
            btnNo.style.left = x + "px";
            btnNo.style.top = y + "px";
            
            // Boton Si crece
            btnSi.style.transform = `scale(${1 + (sizeInc * 0.15)})`;
        });

        // Al aceptar
        btnSi.addEventListener('click', () => {
            contenedor.classList.remove('visible');
            setTimeout(() => {
                contenedor.classList.add('hidden');
                celebracion.classList.remove('hidden');
                setTimeout(() => celebracion.classList.add('visible'), 50);
                rainSpeed = 60; // Lluvia mas rapida al final
            }, 500);
        });

        // Lluvia constante
        const emojis = ['‚ù§Ô∏è', 'üíñ', '‚ú®', 'üå∏', 'üíï'];
        let rainSpeed = 300;

        function createHeart(init = false) {
            const h = document.createElement('div');
            h.classList.add('heart');
            h.innerText = emojis[Math.floor(Math.random() * emojis.length)];
            h.style.left = Math.random() * 100 + 'vw';
            h.style.fontSize = Math.random() * 15 + 15 + 'px';
            if(init) h.style.top = Math.random() * 100 + 'vh';
            
            const dur = Math.random() * 3 + 2;
            h.style.animationDuration = dur + 's';
            
            heartRainContainer.appendChild(h);
            setTimeout(() => h.remove(), dur * 1000);
        }

        // Lluvia inicial
        for(let i=0; i<15; i++) createHeart(true);

        // Bucle de lluvia
        function startRain() {
            createHeart();
            setTimeout(startRain, rainSpeed);
        }
        startRain();
    </script>
